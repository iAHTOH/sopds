##
## Установка на RaspberryPI
## БД SQLITE
##
sudo apt-get update
sudo apt-get install git
sudo apt-get install python3-pip
sudo apt-get install python3-lxml
sudo apt-get install libopenjp2-7-dev
sudo apt-get install libtiff5
cd /home
sudo git clone https://github.com/mitshel/eopds.git
cd /home/eopds
sudo python3 -m pip install -r requirements.txt
sudo python3 manage.py migrate
sudo python3 manage.py eopds_util clear
sudo python3 manage.py createsuperuser
sudo python3 manage.py eopds_util setconf SOPDS_ROOT_LIB /home/eopds/opds_catalog/tests/data
sudo python3 manage.py eopds_util setconf SOPDS_LANGUAGE ru-RU
sudo python3 manage.py eopds_util setconf SOPDS_SCAN_START_DIRECTLY True
sudo python3 manage.py eopds_scanner start --daemon
sudo python3 manage.py eopds_server start --daemon

##
## Настраиваем MYSQL
##
##
sudo apt-get install mysql-server
sudo apt-get install python3-mysqldb
sudo mysql mysql  
mysql > create database if not exists eopds default charset=utf8;  
mysql > grant all on eopds.* to 'eopds'@'localhost' identified by 'eopds';  
mysql > commit;  
mysql > ^C
cd /home/eopds
sudo vi eopds/settings.py

# Убираем комментарии в настройке БД Mysql
DATABASES = {   
    'default': {
        'ENGINE': 'django.db.backends.mysql',
        'NAME': 'eopds',
        'HOST': 'localhost',
        'USER': 'eopds',
        'PASSWORD' : 'eopds',
        'OPTIONS' : {^M
            'init_command': "SET default_storage_engine=MyISAM;\
                             SET sql_mode='';"
        }
    }             
}
